    About

    ng-grid Originally built on knockout we wanted to port it to angular.

    ng-grid is in its first iteration. We are going to be adding more features here as we head to a 2.0 release.

    The bare bones:

    <script type="text/javascript" src="angular.js"></script>
<script type="text/javascript" src="ng-grid.js"></script>
<script>
    angular.module('myApp',['ngGrid', ... {other includes}]);
</script>
<link rel="stylesheet" type="text/css" href="../ng-grid.css" /> 
<body ng-app="myApp">
    <div ng-grid="myOptions"></div>
<body>

    // Define your own controller somewhere..
function MyCtrl($scope) {
    $scope.myData = [{name: "Moroni", age: 50},
                     {name: "Tiancum", age: 43},
                     {name: "Jacob", age: 27},
                     {name: "Nephi", age: 29},
                     {name: "Enos", age: 34}];
    $scope.myOptions = { data: myData };
};
